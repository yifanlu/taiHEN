249  if (ret < 0) {
250  LOG("HEN patches failed: %x", ret);
251  return SCE_KERNEL_START_FAILED;
252  }

254  // TODO: replace this section with configurable options

256  SceKernelLMOption opt;

258

262  ret = sceKernelStartModuleForDriver(henkaku, 0, NULL, 0, NULL, &res);
263  LOG("start: %x, %x", ret, res);
