155 static inline uint32_t next_pow_2(uint32_t v) {
156  v--;
157  v |= v >> 1;
158  v |= v >> 2;
159  v |= v >> 4;
160  v |= v >> 8;
161  v |= v >> 16;
162  v++;
163  v += (v == 0);
164  return v;

167 void slab_init(struct slab_chain *const sch, const size_t itemsize, SceUID pid)
168 {
169  assert(sch != NULL);
170  assert(itemsize >= 1 && itemsize <= SIZE_MAX);
171  assert(POWEROF2(slab_pagesize));
172 
173  sch->itemsize = itemsize;
174  sch->pid = pid;
175 
176  const size_t data_offset = offsetof(struct slab_header, data);
177  const size_t least_slabsize = data_offset + 64 * sch->itemsize;
178  sch->slabsize = (size_t) next_pow_2(least_slabsize);
179  sch->itemcount = 64;
180 
181  if (sch->slabsize - least_slabsize != 0) {
182  const size_t shrinked_slabsize = sch->slabsize >> 1;
183 
184  if (data_offset < shrinked_slabsize &&
185  shrinked_slabsize - data_offset >= 2 * sch->itemsize) {
186 
187  sch->slabsize = shrinked_slabsize;
188  sch->itemcount = (shrinked_slabsize - data_offset) / sch->itemsize;
189  }
190  }
191 
192  sch->pages_per_alloc = sch->slabsize > slab_pagesize ?
193  sch->slabsize : slab_pagesize;
194 
195  sch->empty_slotmask = ~SLOTS_ALL_ZERO >> (64 - sch->itemcount);
196  sch->initial_slotmask = sch->empty_slotmask ^ SLOTS_FIRST;
197  sch->alignment_mask = ~(sch->slabsize - 1);
198  sch->partial = sch->empty = sch->full = NULL;
199 
200  assert(slab_is_valid(sch));
201 }

203 void *slab_alloc(struct slab_chain *const sch, uintptr_t *exe_addr)
204 {
205  assert(sch != NULL);
206  assert(slab_is_valid(sch));
207 
208  if (LIKELY(sch->partial != NULL)) {
209  /* found a partial slab, locate the first free slot */
210  register const size_t slot = FIRST_FREE_SLOT(sch->partial->slots);
211  sch->partial->slots ^= SLOTS_FIRST << slot;
212 
213  if (UNLIKELY(sch->partial->slots == SLOTS_ALL_ZERO)) {
214  /* slab has become full, change state from partial to full */
215  struct slab_header *const tmp = sch->partial;
216 
217  /* skip first slab from partial list */
218  if (LIKELY((sch->partial = sch->partial->next) != NULL))
219  sch->partial->prev = NULL;
220 
221  if (LIKELY((tmp->next = sch->full) != NULL))
222  sch->full->prev = tmp;
223 
224  sch->full = tmp;
225  *exe_addr = sch->full->exe_data + slot * sch->itemsize;
226  return sch->full->data + slot * sch->itemsize;
227  } else {
228  *exe_addr = sch->partial->exe_data + slot * sch->itemsize;
229  return sch->partial->data + slot * sch->itemsize;
230  }
231  } else if (LIKELY((sch->partial = sch->empty) != NULL)) {
232  /* found an empty slab, change state from empty to partial */
233  if (LIKELY((sch->empty = sch->empty->next) != NULL))
234  sch->empty->prev = NULL;
235 
236  sch->partial->next = NULL;
237 
238  /* slab is located either at the beginning of page, or beyond */
239  UNLIKELY(sch->partial->refcount != 0) ?
240  sch->partial->refcount++ : sch->partial->page->refcount++;
241 
242  sch->partial->slots = sch->initial_slotmask;
243  *exe_addr = sch->partial->exe_data;
244  return sch->partial->data;
245  } else {
246  /* no empty or partial slabs available, create a new one */
247  SceUID write_res, exe_res;
248  uintptr_t exe_data;
249  if ((write_res = sce_exe_alloc(sch->pid, (void **)&sch->partial, &exe_data,
250  &exe_res, sch->slabsize, sch->pages_per_alloc)) < 0) {
251  *exe_addr = 0;
252  return sch->partial = NULL;
253  }
254  sch->partial->write_res = write_res;
255  sch->partial->exe_res = exe_res;
256  sch->partial->exe_data = exe_data + offsetof(struct slab_header, data);
257  exe_data += sch->slabsize;
258 
259  struct slab_header *prev = NULL;
260 
261  const char *const page_end =
262  (char *) sch->partial + sch->pages_per_alloc;
263 
264  union {
265  const char *c;
266  struct slab_header *const s;
267  } curr = {
268  .c = (const char *) sch->partial + sch->slabsize
269  };
270 
271  __builtin_prefetch(sch->partial, 1);
272 
273  sch->partial->prev = sch->partial->next = NULL;
274  sch->partial->refcount = 1;
275  sch->partial->slots = sch->initial_slotmask;
276 
277  if (LIKELY(curr.c != page_end)) {
278  curr.s->prev = NULL;
279  curr.s->refcount = 0;
280  curr.s->page = sch->partial;
281  curr.s->write_res = write_res;
282  curr.s->exe_res = exe_res;
283  curr.s->exe_data = exe_data;
284  exe_data += sch->slabsize;
285  curr.s->slots = sch->empty_slotmask;
286  sch->empty = prev = curr.s;
287 
288  while (LIKELY((curr.c += sch->slabsize) != page_end)) {
289  prev->next = curr.s;
290  curr.s->prev = prev;
291  curr.s->refcount = 0;
292  curr.s->page = sch->partial;
293  curr.s->write_res = write_res;
294  curr.s->exe_res = exe_res;
295  curr.s->exe_data = exe_data;
296  exe_data += sch->slabsize;
297  curr.s->slots = sch->empty_slotmask;
298  prev = curr.s;
299  }
300 
301  prev->next = NULL;
302  }
303 
304  *exe_addr = sch->partial->exe_data;
305  return sch->partial->data;
306  }
307 
308  /* unreachable */
309 }

311 void slab_free(struct slab_chain *const sch, const void *const addr)
312 {
313  assert(sch != NULL);
314  assert(slab_is_valid(sch));
315  assert(addr != NULL);
316 
317  struct slab_header *const slab = (void *)
318  ((uintptr_t) addr & sch->alignment_mask);
319 
320  register const int slot = ((char *) addr - (char *) slab -
321  offsetof(struct slab_header, data)) / sch->itemsize;
322 
323  if (UNLIKELY(slab->slots == SLOTS_ALL_ZERO)) {
324  /* target slab is full, change state to partial */
325  slab->slots = SLOTS_FIRST << slot;
326 
327  if (LIKELY(slab != sch->full)) {
328  if (LIKELY((slab->prev->next = slab->next) != NULL))
329  slab->next->prev = slab->prev;
330 
331  slab->prev = NULL;
332  } else if (LIKELY((sch->full = sch->full->next) != NULL)) {
333  sch->full->prev = NULL;
334  }
335 
336  slab->next = sch->partial;
337 
338  if (LIKELY(sch->partial != NULL))
339  sch->partial->prev = slab;
340 
341  sch->partial = slab;
342  } else if (UNLIKELY(ONE_USED_SLOT(slab->slots, sch->empty_slotmask))) {
343  /* target slab is partial and has only one filled slot */
344  if (UNLIKELY(slab->refcount == 1 || (slab->refcount == 0 &&
345  slab->page->refcount == 1))) {
346 
347  /* unmap the whole page if this slab is the only partial one */
348  if (LIKELY(slab != sch->partial)) {
349  if (LIKELY((slab->prev->next = slab->next) != NULL))
350  slab->next->prev = slab->prev;
351  } else if (LIKELY((sch->partial = sch->partial->next) != NULL)) {
352  sch->partial->prev = NULL;
353  }
354 
355  void *const page = UNLIKELY(slab->refcount != 0) ? slab : slab->page;
356  const char *const page_end = (char *) page + sch->pages_per_alloc;
357  char found_head = 0;
358 
359  union {
360  const char *c;
361  const struct slab_header *const s;
362  } s;
363 
364  for (s.c = page; s.c != page_end; s.c += sch->slabsize) {
365  if (UNLIKELY(s.s == sch->empty))
366  found_head = 1;
367  else if (UNLIKELY(s.s == slab))
368  continue;
369  else if (LIKELY((s.s->prev->next = s.s->next) != NULL))
370  s.s->next->prev = s.s->prev;
371  }
372 
373  if (UNLIKELY(found_head && (sch->empty = sch->empty->next) != NULL))
374  sch->empty->prev = NULL;
375 
376  sce_exe_free(slab->write_res, slab->exe_res);
377  } else {
378  slab->slots = sch->empty_slotmask;
379 
380  if (LIKELY(slab != sch->partial)) {
381  if (LIKELY((slab->prev->next = slab->next) != NULL))
382  slab->next->prev = slab->prev;
383 
384  slab->prev = NULL;
385  } else if (LIKELY((sch->partial = sch->partial->next) != NULL)) {
386  sch->partial->prev = NULL;
387  }
388 
389  slab->next = sch->empty;
390 
391  if (LIKELY(sch->empty != NULL))
392  sch->empty->prev = slab;
393 
394  sch->empty = slab;
395 
396  UNLIKELY(slab->refcount != 0) ?
397  slab->refcount-- : slab->page->refcount--;
398  }
399  } else {
400  /* target slab is partial, no need to change state */
401  slab->slots |= SLOTS_FIRST << slot;
402  }
403 }

418 void slab_traverse(const struct slab_chain *const sch, void (*fn)(const void *))
419 {
420  assert(sch != NULL);
421  assert(fn != NULL);
422  assert(slab_is_valid(sch));
423 
424  const struct slab_header *slab;
425  const char *item, *end;
426  const size_t data_offset = offsetof(struct slab_header, data);
427 
428  for (slab = sch->partial; slab; slab = slab->next) {
429  item = (const char *) slab + data_offset;
430  end = item + sch->itemcount * sch->itemsize;
431  uint64_t mask = SLOTS_FIRST;
432 
433  do {
434  if (!(slab->slots & mask))
435  fn(item);
436 
437  mask <<= 1;
438  } while ((item += sch->itemsize) != end);
439  }
440 
441  for (slab = sch->full; slab; slab = slab->next) {
442  item = (const char *) slab + data_offset;
443  end = item + sch->itemcount * sch->itemsize;
444 
445  do fn(item);
446  while ((item += sch->itemsize) != end);
447  }
